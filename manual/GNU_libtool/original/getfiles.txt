#------------------------------
# ソース入手と tarball 伸長
#------------------------------
$ cd ~/src
$ wget -N https://ftpmirror.gnu.org/libtool/libtool-2.4.6.tar.gz
$ tar xf libtool-2.4.6.tar.gz
$ ls
libtool-2.4.6

#------------------------------
# 当プロジェクトへの man ページのコピー
#------------------------------

$ cd $(JMTOP)/manual/GNU_libtool/original
$ cat > getfiles.sh <<"EOF"
#!/bin/sh

SRCDIR=~/src/libtool-2.4.6

rm -fr man1
mkdir  man1
cp -p $SRCDIR/doc/*.1 man1
EOF

$ sh getfiles.sh
