#------------------------------
# ソース tarball 入手と伸長
#------------------------------
$ cd ~/src
$ wget -N https://github.com/mesonbuild/meson/releases/download/0.60.3/meson-0.60.3.tar.gz
$ tar xf meson-0.60.3.tar.gz
$ ls
meson-0.60.3

#------------------------------
# 当プロジェクトへの man ページのコピー
#------------------------------

$ cd $(JMTOP)/manual/meson/original
$ cat > getfiles.sh <<"EOF"
#!/bin/sh

SRCDIR=~/src/meson-0.60.3

rm -fr man1
mkdir  man1

cp -p $SRCDIR/man/*.1 man1
EOF

$ sh getfiles.sh
